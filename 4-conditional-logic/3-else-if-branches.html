else if branches let you create as many different options as you want. The program only chooses one of the options in the entire statement. The following code pops up different alerts depending on the value of number.

  var number = 0;

  if (number === 0) {
    alert('equals 0');
  } else if (number === 1) {
    alert('equals 1');
  } else if (number === 2) {
    alert('equals 2');
  }
