You can use the && operator to get sprites to jump. Just check if the up key is getting pressed and the player is touching the ground.

You can check if a sprite is touching the ground with:

  someSprite.body.touching.down

Here it is in context:

  ...

  var upKey;

  function createFunction() {
    ...

    upKey = game.input.keyboard.addKey(Phaser.Keyboard.UP);
  }

  function updateFunction() {
    ...

    if (leftKey.isDown) {
      player.body.velocity.x = -50;
    } else if (rightKey.isDown){
      player.body.velocity.x = 50;
    } else if (upKey.isDown && player.body.touching.down) {
      player.body.velocity.y = -100;
    } else {
      player.body.velocity.x = 0;
    }
  }
