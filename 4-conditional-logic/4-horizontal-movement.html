You can use if...else statements plus a few other tools to get sprites to move on the screen.

You can add a input key to the game with:

  someKeyName = game.input.keyboard.addKey(Phaser.Keyboard.SOMEDIRECTION)

You can check if a key is pressed with:

  someKeyName.isDown

You can change the velocity of a sprite with:

  someSprite.body.velocity.x = someNumber;

Here's all of that in context:

  ...

  // creates a variables for the left arrow key
  var leftKey;

  function createFunction() {
    ...
    ground3.body.immovable = true;

    // assigns the left arrow key to the leftKey variable
    leftKey = game.input.keyboard.addKey(Phaser.Keyboard.LEFT);

  }

  function updateFunction() {
    ...
    game.physics.arcade.collide(player, ground3);

    // checks if the left key is pressed
    if (leftKey.isDown) {
      // changes the players horizontal velocity to -50
      player.body.velocity.x = -50;
    } else {
      player.body.velocity.x = 0;
    }

}
